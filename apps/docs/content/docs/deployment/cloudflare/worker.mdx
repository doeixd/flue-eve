# Worker Setup

```ts
import { createEveWorkerApp } from '@flue-eve/compat-server/worker';

export default createEveWorkerApp({
  agentName: 'assistant',
  persistence: {
    type: 'kv',
    kvNamespace: MY_KV,
  },
});
```

## Local development

```bash
npm run dev
# or
wrangler dev
```

## Deploy

```bash
npm run deploy
```
